# Backup erstellen
mariadb-dump --single-transaction --default-character-set=utf8mb4 --no-tablespaces --user='nextcloud_dbu' --password='<password>' 'nextcloud_dba' > nextcloud-sqlbkp_`date +"%Y%m%d"`.bak

## Backup wiederherstellen
sudo mariadb -h localhost -u nextcloud_dbu -p'Password' nextcloud_dba < nextcloud-sqlbkp_<Datum>.bak
