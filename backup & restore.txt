# Backup erstellen

# Mariadb Backup
mkdir -p /home/marcel/Nextcloud-backup/dbdumps
mariadb-dump --single-transaction --default-character-set=utf8mb4 --no-tablespaces --user='nextcloud_dbu' --password='<password>' 'nextcloud_dba' > /home/marcel/Nextcloud-backup/dbdumps/nextcloud-sqlbkp_`date +"%Y%m%d"`.bak

Daten Backup
mkdir -p /home/marcel/Nextcloud-backup/nextcloud/nextcloud_`date +"%Y%m%d"`.bak/
sudo rsync -Aavx /srv/www/nextcloud/ /home/marcel/Nextcloud-backup/nextcloud/nextcloud_`date +"%Y%m%d"`.bak/
mkdir -p /home/marcel/Nextcloud-backup/data/data_`date +"%Y%m%d"`.bak/
sudo rsync -Aavx /home/data/ /home/marcel/Nextcloud-backup/data/data_`date +"%Y%m%d"`.bak/

sudo chown -R marcel /home/marcel/Nextcloud-backup/data/data_`date +"%Y%m%d"`.bak/

## Backup nach Extern kopieren

mkdir -p /home/marcel/Dokumente/Nextcloud-backup/
sudo scp -r marcel@87.106.166.194:/home/marcel/Nextcloud-backup/ /home/marcel/Dokumente/Nextcloud-backup_`date +"%Y%m%d"`.bak/

## Backup zurück kopieren
# Remote
mkdir -p /home/marcel/Nextcloud-stage/
# Lokal
sudo scp -r /home/marcel/Dokumente/Nextcloud-backup_`date +"%Y%m%d"`.bak/ marcel@87.106.166.194:/home/marcel/Nextcloud-stage/Nextcloud-backup_`date +"%Y%m%d"`.bak/


## Backup wiederherstellen
sudo mariadb -h localhost -u nextcloud_dbu -p'Password' nextcloud_dba < /home/marcel/Nextcloud-backup_`date +"%Y%m%d"`.bak/dbdumps/nextcloud-sqlbkp_`date +"%Y%m%d"`.bak/

sudo rsync -Aavx /home/marcel/Nextcloud-backup_`date +"%Y%m%d"`.bak/nextcloud/nextcloud_`date +"%Y%m%d"`.bak/ /srv/www/nextcloud/

sudo rsync -Aavx /home/marcel/Nextcloud-backup_`date +"%Y%m%d"`.bak/data/data_`date +"%Y%m%d"`.bak/ /home/data/
